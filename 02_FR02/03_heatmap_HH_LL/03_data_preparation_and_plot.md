# Prepare data layers for the FINRISK heatmap figure

We need to prepare the data layers and plot a  multi-panel FINRISK heatmap.

Generated by this script:
- `VF_ab_matrix_filt_heatmap.RData`: VF-abundance matrix for the heatmap (HH/LL subset, log10(RPKM + 1)); restricted to participants free of prevalent LD + complete covariates, and to VFs detected in ≥5% of that subset
- `VF_category.csv`: VF category annotation
- `ZicoSeq_results_FINRISK.csv`: ZicoSeq differential-abundance results (HH vs LL)
- `Taxa_ab_matrix_top5_heatmap.RData`: top-5 species abundance matrix for the right-hand panel

Pre-existing input (not generated here):
- `VF_Ecoli_preval.csv` : with % VF prevalence in E. coli genomes(GTDB), already built on the Baker cluster from `gtdb_r214_vfdb_setA_diamond_70ident_90scov_filtered_hits.Rdata`

Inputs:
- `FINRISK_d0_VF_modelling.RData` (from `02_02_run_models_incidence.md`) -> `d0`: modelling data (HH/LL groups, covariates)
- `vfdb_hits_normalized_VF_level.RData`: per-participant VF abundance
- `phyloseq.rds`: to get relative abundances

```R
# ==============================================================================
# 0. Global configuration & libraries
# ==============================================================================
library(tidyverse)
library(phyloseq)
#remotes::install_local("dirmult_0.1.3-5.tar.gz")
#remotes::install_local("GUniFrac_1.8.tar.gz")
library(GUniFrac)
packageVersion("GUniFrac")  # 1.8

options(width = 200, stringsAsFactors = FALSE)

OUTDIR <- "03_heatmap_HH_LL"
dir.create(OUTDIR, showWarnings = FALSE)

# ==============================================================================
# 1. Load and prepare modelling data and VF hits
# ==============================================================================

load("/home/camigazo/02_VF_aggregated_analysis/FINRISK_d0_VF_modelling.RData")
dim(d0)  # 7226 x 29

load("/home/camigazo/data/vfdb_hits_normalized.RData")
vf_hits <- diamond_final; rm(diamond_final)
dim(vf_hits)  # 1007163 x 11

# HH/LL subset: no prevalent LD + complete covariates
HL_LL_ind <- d0 %>%
  filter(group2 %in% c("HH", "LL")) %>%
  filter(PREVAL_LIVERDIS == 0) %>%
  drop_na(BL_AGE, MEN, BMI, ALKI2_FR02, BL_USE_RX_J01, ReadDepthMillions, PREVAL_DIAB) %>%
  pull(Barcode)

length(HL_LL_ind)  # 1837, correct

d0_filt <- d0 %>% filter(Barcode %in% HL_LL_ind)
dim(d0_filt)  # 1837 x 29

# ==============================================================================
# 2. Aggregate gene-level hits to VF level, then prevalence-filter
#    log10(sum RPKM + 1); restricted to the HH/LL subset; VFs in >= 5% of subset
# ==============================================================================
PREV_THRESHOLD <- 0.05

vf_hits <- vf_hits %>%
  filter(SampleID %in% HL_LL_ind) %>%
  group_by(SampleID, VFID, VF_Name, VFcategory) %>%
  summarise(Reads = sum(NumberOfHighQualityReads), RPKM = sum(RPKM_HighQuality), .groups = "drop") %>%
  filter(Reads > 0)

prevalence <- vf_hits %>%
  distinct(SampleID, VFID) %>%
  count(VFID, name = "Detection") %>%
  mutate(Prevalence = 100 * Detection / length(HL_LL_ind))

selected_VFs <- prevalence %>% filter(Prevalence >= 100 * PREV_THRESHOLD) %>% pull(VFID)
length(selected_VFs)  # 33

# ==============================================================================
# 3. VF abundance matrix -> VF_ab_matrix_filt_heatmap.RData
#    log10(RPKM + 1), samples x VFs
# ==============================================================================
m <- vf_hits %>%
  filter(VFID %in% selected_VFs) %>%
  transmute(SampleID, VFID, value = log10(RPKM + 1)) %>%
  # force every person in HL_LL_ind to appear, even with no selected-VF hits
  mutate(SampleID = factor(SampleID, levels = HL_LL_ind)) %>%
  complete(SampleID, VFID, fill = list(value = 0)) %>%      # <<< adds the missing rows
  pivot_wider(names_from = VFID, values_from = value, values_fill = 0) %>%
  column_to_rownames("SampleID")

setdiff(HL_LL_ind, rownames(m))  # no missing person in the matrix
cat(sprintf("matrix: %d samples x %d VFs\n", nrow(m), ncol(m)))
# matrix: 1837 samples x 33 VFs

m <- as.matrix(m)

save(m, file = file.path(OUTDIR, "VF_ab_matrix_filt_heatmap.RData"))

# ==============================================================================
# 4. VF category annotation -> VF_category.csv
# ==============================================================================
VF_category <- vf_hits %>%
  select(VFID, VFcategory) %>%
  distinct() %>%
  filter(VFID %in% selected_VFs)

dim(VF_category)  # 33 x 2

write.csv(VF_category, file = file.path(OUTDIR, "VF_category.csv"))

# ==============================================================================
# 5. ZicoSeq: differential species abundance between HH and LL richness–burden
#    -> ZicoSeq_results_FINRISK.csv
# ==============================================================================
PREV_FILT <- 0.05

# Basic covariate set, include bath
COVARIATES <- c("BL_AGE", "MEN", "BMI", "BL_USE_RX_J01", "ALKI2_FR02", "ReadDepthMillions", "PREVAL_DIAB", "batch_name")

# Microbiome data
ps <- readRDS("/home/camigazo/data/phyloseq.rds")
ps <- subset_taxa(ps, Domain == "Bacteria") %>%
  prune_taxa(taxa_sums(.) > 0, .) # # remove taxa with no reads too
ps # 8244 taxa x 7226 samples

# Extract count matrix and filter for individuals with metadata
feat <- otu_table(ps) %>%
  as.data.frame() %>%
  as.matrix()
feat <- feat[, d0_filt$Barcode]
feat <- feat[rowSums(feat) > 0, ]
dim(feat) # 7115 x 1837 rows = bac, cols = samples (Barcode)

# Align metadata to feature-matrix column order
metadata_zico <- d0_filt
rownames(metadata_zico) <- metadata_zico$Barcode
metadata_zico <- metadata_zico[colnames(feat), ]
stopifnot(all(colnames(feat) == metadata_zico$Barcode))
cat(sprintf("ZicoSeq input: %d features x %d samples\n", nrow(feat), ncol(feat)))
# ZicoSeq input: 7115 features x 1837 samples

nrow(metadata_zico) * PREV_FILT  # prevalence-filter cutoff (n individuals) -> 91.85

# Run ZicoSeq
set.seed(123)
ZicoSeq.obj <- ZicoSeq(
  meta.dat = as.data.frame(metadata_zico),
  feature.dat = feat,
  grp.name = "group2",
  adj.name = COVARIATES,
  feature.dat.type = "count",
  prev.filter = PREV_FILT, mean.abund.filter = 0, max.abund.filter = 0.002,
  min.prop = 0, is.winsor = TRUE, outlier.pct = 0.03, winsor.end = "top",
  is.post.sample = TRUE, post.sample.no = 25,
  link.func = list(function(x) x^0.5), stats.combine.func = max,
  perm.no = 200, strata = NULL, ref.pct = 0.5, stage.no = 6, excl.pct = 0.2,
  is.fwer = TRUE, verbose = TRUE, return.feature.dat = TRUE
)
# 6066  features are filtered!
# The data has  1837  samples and  1049  features will be tested!
# On average,  55  outlier counts will be replaced for each feature!

# Extract results, verify direction, and write
# helper: coerce a slot to a named vector keyed by feature, regardless of shape
as_feat_vec <- function(x) {
  if (is.matrix(x)) {
    v <- x[, 1]                 # single link.func -> one column (Func1)
    names(v) <- rownames(x)
    v
  } else x                      # already a named vector
}

p_raw <- as_feat_vec(ZicoSeq.obj$p.raw)
fdr   <- as_feat_vec(ZicoSeq.obj$p.adj.fdr)
r2    <- as_feat_vec(ZicoSeq.obj$R2 * 100)
feats <- names(fdr)

res_all <- tibble(
  taxa       = feats,
  p_raw      = p_raw[feats],
  FDR        = fdr[feats],
  R2         = r2[feats],
  prevalence = rowMeans(ZicoSeq.obj$feature.dat[feats, , drop = FALSE] > 0)
)

# Signed direction from the adjusted feature data: mean in HH vs LL
hh <- metadata_zico$group2 == "HH"
dir_sign <- apply(ZicoSeq.obj$feature.dat[res_all$taxa, , drop = FALSE], 1,
                  function(x) sign(mean(x[hh]) - mean(x[!hh])))

res_all <- res_all %>% mutate(direction = ifelse(dir_sign > 0, "Elevated", "Decreased"))  # Elevated = higher in HH

write.csv(res_all, file = file.path(OUTDIR, "ZicoSeq_results_FINRISK.csv"))

# ==============================================================================
# 6. Top-5 species abundance matrix -> Taxa_ab_matrix_top5_heatmap.RData
#    Right-hand species panel: top 5 by R2 among ZicoSeq FDR < 0.05 hits
# ==============================================================================

# Top 5 differentially abundant species (defines the right-hand panel)
top5 <- res_all %>%
  filter(FDR < 0.05) %>%
  slice_max(R2, n = 5) 

top5_annotation <- top5 %>% transmute(taxa, R2, prevalence)
write.csv(top5_annotation, file = file.path(OUTDIR, "Taxa_top5_annotation.csv"), row.names = FALSE)

top5_abrel <- transform_sample_counts(ps, function(x) x / sum(x)) %>%
  subset_taxa(Species %in% top5$taxa) %>%  
  psmelt() %>%
  mutate(Abundance=Abundance*100) %>%
  filter(Sample %in% HL_LL_ind)

taxa_mat <- top5_abrel %>%
  select(OTU, Sample, Abundance) %>%
  pivot_wider(names_from = OTU, values_from = Abundance, values_fill = 0) %>% # samples absent for a species -> 0
  column_to_rownames("Sample") %>%
  as.matrix()

stopifnot(setequal(rownames(taxa_mat), rownames(m)))
taxa_mat <- taxa_mat[rownames(m), , drop = FALSE] # align rows to m

cat(sprintf("taxa matrix: %d samples x %d species\n", nrow(taxa_mat), ncol(taxa_mat)))
# taxa matrix: 1837 samples x 5 species

# Saved object is `taxa_mat` (the VF matrix in section 3 is `m`)
save(taxa_mat, file = file.path(OUTDIR, "Taxa_ab_matrix_top5_heatmap.RData"))
```

# Multi-panel VF heatmap

```R
# ---- 0. Libraries & options --------------------------------------------------
library(tidyverse)
library(ComplexHeatmap)
library(circlize)

ht_opt$message <- FALSE
options(width = 200, stringsAsFactors = FALSE)

OUTDIR <- "03_heatmap_HH_LL"

# ---- 1. Load inputs ----------------------------------------------------------
load(file.path(OUTDIR, "VF_ab_matrix_filt_heatmap.RData"))     # m        : participants x VFs (log10(RPKM+1))
load(file.path(OUTDIR, "Taxa_ab_matrix_top5_heatmap.RData"))   # taxa_mat : participants x 5 species (percent)

max(taxa_mat) # 95

vf_cat_df <- read_csv(file.path(OUTDIR, "VF_category.csv"),         show_col_types = FALSE)
ecoli_df  <- read_csv(file.path(OUTDIR, "VF_Ecoli_preval.csv"),     show_col_types = FALSE)
top5      <- read_csv(file.path(OUTDIR, "Taxa_top5_annotation.csv"),show_col_types = FALSE)

### load FINRISK d0 and key the VF group on Barcode 
load("/home/camigazo/02_VF_aggregated_analysis/FINRISK_d0_VF_modelling.RData")   # d0
group_lookup <- with(d0, setNames(as.character(dplyr::recode(group2, HH = "High", LL = "Low")), Barcode))

# ---- 2. ID-keyed lookups (survive any reordering of m's columns) -------------
ecoli_prev_lookup <- setNames(ecoli_df$ecoli_prevalence_pct, ecoli_df$vfid)
vf_cat_lookup     <- setNames(vf_cat_df$VFcategory,          vf_cat_df$VFID)

# ---- 3. Align participants across the two matrices ---------------------------
stopifnot(setequal(rownames(taxa_mat), rownames(m)))
taxa_mat <- taxa_mat[rownames(m), , drop = FALSE]
n_part   <- nrow(m)

# ---- 4. Fixed orderings ------------------------------------------------------
# 4a. Locate the E. coli column
ecoli_taxon <- top5$taxa[1]

# 4b. Participant rows: E. coli abundance high -> low.
ecoli_ab  <- taxa_mat[, ecoli_taxon]
row_order <- order(ecoli_ab, rowSums(m), decreasing = TRUE)
m         <- m[row_order, , drop = FALSE]
taxa_mat  <- taxa_mat[row_order, , drop = FALSE]

# 4c. VF columns: ordered by hierarchical clustering inside ht1 (dendrogram
#     hidden) -> no manual column ordering here.

# 4d. Species columns: R2 high -> low
sp_order <- top5$taxa[order(top5$R2, decreasing = TRUE)]
sp_order <- sp_order[sp_order %in% colnames(taxa_mat)]
taxa_mat <- taxa_mat[, sp_order, drop = FALSE]

# ---- 5. Annotation vectors (re-aligned to the FINAL row/column order) --------
vf_group   <- group_lookup[rownames(m)];      stopifnot(!anyNA(vf_group))
vf_group_f <- factor(vf_group, levels = c("High", "Low"))   # slice order
n_high     <- sum(vf_group == "High")
n_low      <- sum(vf_group == "Low")
ecoli_prev <- ecoli_prev_lookup[colnames(m)]; ecoli_prev[is.na(ecoli_prev)] <- 0
vf_cat     <- vf_cat_lookup[colnames(m)];      stopifnot(!anyNA(vf_cat))

vf_cat_cols <- c(
  "Adherence"                                    = "#F2978D",
  "Antimicrobial activity/Competitive advantage" = "#5E60CE",
  "Biofilm"                                       = "#F4D35E",
  "Effector delivery system"                      = "#19A39A",
  "Exotoxin"                                       = "#EA6A2E",
  "Immune modulation"                             = "#000000",
  "Invasion"                                       = "#9E9E9E",
  "Motility"                                       = "#9BE08C",
  "Nutritional/Metabolic factor"                  = "#CDB592",
  "Regulation"                                     = "#2E8B57",
  "Stress survival"                               = "#6BAED6"
)
vf_cat <- factor(vf_cat, levels = names(vf_cat_cols))

# Species column labels: "Species (R2=.., n=..)"
sp_label        <- with(top5, sprintf("%s (R2=%.1f, n=%d)", taxa, R2, round(prevalence * n_part)))
names(sp_label) <- top5$taxa
col_labels      <- sp_label[colnames(taxa_mat)]; stopifnot(!anyNA(col_labels))

# ---- 6. Colour scales --------------------------------------------------------

# E coli / HH x LL
ecoli_col <- colorRamp2(c(0, 100), c("white", "#CB181D"))
grp_col   <- c(High = "#E0827B", Low = "#6B5B95")

# VF panel
max(m) # 3.4
vf_breaks <- c(0, 1, 2, 3, 4) 
vf_col <- colorRamp2(vf_breaks, c("white","#ABD9E9","#FDAE61","#E34A33","#B30000"))

# Species panel
taxa_mat <- log10(taxa_mat+1)
max(taxa_mat) # 1.98
sp_breaks <- log10(c(0, 1, 10, 50, 95) + 1)
taxa_col  <- colorRamp2(sp_breaks, c("white","#ABD9E9","#FDAE61","#E34A33","#B30000"))

# ---- 7. Annotations ----------------------------------------------------------
# VF-panel top tracks: % in E. coli (white -> red) over VF category
top_anno <- HeatmapAnnotation(
  `% in E. coli genomes` = ecoli_prev,
  `VF category`          = vf_cat,
  col = list(`% in E. coli genomes` = ecoli_col, `VF category` = vf_cat_cols),
  border = TRUE,
  annotation_name_side = "left",
  annotation_name_gp   = gpar(fontsize = 8),
  simple_anno_size     = unit(3, "mm"),
  gap = unit(1, "mm"),
  annotation_legend_param = list(
    `% in E. coli genomes` = list(at = c(0, 50, 100), legend_direction = "horizontal")
  )
)

# VF-group strip -> LEFT of the plot (annotation only, not used for ordering)
left_anno <- rowAnnotation(
  `VF group` = vf_group,
  col = list(`VF group` = grp_col),
  border = TRUE,
  annotation_name_side = "top", annotation_name_rot = 0,
  annotation_name_gp   = gpar(fontsize = 8),
  width = unit(4, "mm")
)

# ---- 8. Heatmaps -------------------------------------------------------------
ht1 <- Heatmap(
  m,
  name = "log10(RPKM+1)",
  col  = vf_col,
  border = TRUE,
  row_split = vf_group_f, cluster_rows = FALSE, cluster_row_slices = FALSE,
  row_gap = unit(2, "mm"),
  cluster_columns = TRUE, show_column_dend = FALSE,  
  show_row_names = FALSE, show_column_names = FALSE,
  top_annotation  = top_anno,
  left_annotation = left_anno,
  row_title = c(sprintf("High VF (n = %s)", format(n_high, big.mark = ",")),
                sprintf("Low VF (n = %s)",  format(n_low,  big.mark = ","))),
  row_title_side = "left", row_title_gp = gpar(fontsize = 11),
  column_title = sprintf("Virulence factors (n = %d)", ncol(m)),
  column_title_side = "bottom", column_title_gp = gpar(fontsize = 11),
  width = unit(9, "cm"),
  use_raster = TRUE, raster_quality = 3,
  heatmap_legend_param = list(
    legend_width     = unit(4, "cm"),
    legend_direction = "horizontal"
  )
)

# ht2: species panel. Inherits ht1's row order; columns fixed by R2 (no dendrogram).
ht2 <- Heatmap(
  taxa_mat,
  name   = "log10(relative abundance + 1)",
  col    = taxa_col,
  border = TRUE,
  cluster_rows = FALSE, cluster_columns = FALSE,
  column_names_side = "top", column_names_rot = 90,
  column_labels = col_labels,
  column_names_gp = gpar(fontface = "italic", fontsize = 8),
  column_names_max_height = unit(4.5, "cm"),
  show_row_names = FALSE,
  width = unit(4.5, "cm"),
  use_raster = TRUE, raster_quality = 3,
  heatmap_legend_param = list(
    legend_direction = "horizontal"
  )
)

# ---- 9. Draw -----------------------------------------------------------------
cairo_pdf(file.path(OUTDIR, "Figure_FINRISK_VF_heatmap.pdf"), width = 7, height = 6)
draw(ht1 + ht2,
     heatmap_legend_side = "bottom", annotation_legend_side = "bottom",
     merge_legends = TRUE, ht_gap = unit(6, "mm"))
dev.off()

# Editable text (Inkscape) -> svglite keeps <text>; raster cells stay raster
library(svglite)
svglite(file.path(OUTDIR, "Figure_FINRISK_VF_heatmap.svg"), width = 7, height = 6)
draw(ht1 + ht2,
     heatmap_legend_side = "bottom", annotation_legend_side = "bottom",
     merge_legends = TRUE, ht_gap = unit(6, "mm"))
dev.off()
```
